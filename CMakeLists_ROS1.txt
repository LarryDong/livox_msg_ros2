cmake_minimum_required(VERSION 3.8)
project(livox_ros_driver)

find_package(catkin REQUIRED COMPONENTS
  std_msgs
  message_generation
)

# 生成消息
add_message_files(
  FILES
  CustomPoint.msg
  CustomMsg.msg
)

# 设置消息依赖
generate_messages(
  DEPENDENCIES
  std_msgs
)

# 导出运行时依赖
catkin_package(
  CATKIN_DEPENDS 
  message_runtime 
  std_msgs
  roscpp
  rospy
)
